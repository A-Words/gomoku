```python
"""Board module: 15x15 grid, place/undo/reset, coordinate conversion."""

BOARD_SIZE = 15
EMPTY = 0
BLACK = 1
WHITE = 2


class Board:
    def __init__(self):
        self.reset()

    def reset(self):
        self.grid = [[EMPTY] * BOARD_SIZE for _ in range(BOARD_SIZE)]
        self.move_history = []  # list of (row, col, color)

    def place(self, row, col, color):
        if not self.is_valid_move(row, col):
            return False
        self.grid[row][col] = color
        self.move_history.append((row, col, color))
        return True

    def undo(self):
        if not self.move_history:
            return None
        row, col, color = self.move_history.pop()
        self.grid[row][col] = EMPTY
        return (row, col, color)

    def is_valid_move(self, row, col):
        if row < 0 or row >= BOARD_SIZE or col < 0 or col >= BOARD_SIZE:
            return False
        return self.grid[row][col] == EMPTY

    def is_full(self):
        for row in self.grid:
            for cell in row:
                if cell == EMPTY:
                    return False
        return True

    def get(self, row, col):
        if 0 <= row < BOARD_SIZE and 0 <= col < BOARD_SIZE:
            return self.grid[row][col]
        return None

    def last_move(self):
        if self.move_history:
            return self.move_history[-1][:2]  # (row, col)
        return None

    def move_count(self):
        return len(self.move_history)
```

```python
"""Win/draw detection for Gomoku."""

from board import BOARD_SIZE, EMPTY

DIRECTIONS = [(0, 1), (1, 0), (1, 1), (1, -1)]  # horizontal, vertical, 2 diagonals


def check_win(grid, row, col):
    color = grid[row][col]
    if color == EMPTY:
        return None

    for dr, dc in DIRECTIONS:
        count = 1
        # Count forward
        r, c = row + dr, col + dc
        while 0 <= r < BOARD_SIZE and 0 <= c < BOARD_SIZE and grid[r][c] == color:
            count += 1
            r += dr
            c += dc
        # Count backward
        r, c = row - dr, col - dc
        while 0 <= r < BOARD_SIZE and 0 <= c < BOARD_SIZE and grid[r][c] == color:
            count += 1
            r -= dr
            c -= dc
        if count >= 5:
            return color

    return None


def check_draw(grid):
    for row in grid:
        for cell in row:
            if cell == EMPTY:
                return False
    return True
```

```python
import os

WINDOW_W = 800
WINDOW_H = 700
BOARD_X = 30
BOARD_Y = 60
BOARD_PX = 560
CELL_SIZE = BOARD_PX // (BOARD_SIZE - 1)
TOP_BAR_H = 50
BOTTOM_BAR_H = 50

# Colors
COLOR_BG = (222, 184, 104)
COLOR_LINE = (92, 74, 30)
COLOR_STAR = (92, 74, 30)
COLOR_BLACK_PIECE = (30, 30, 30)
COLOR_BLACK_HIGHLIGHT = (80, 80, 80)
COLOR_WHITE_PIECE = (240, 240, 235)

COLOR_TEXT_LIGHT = (255, 255, 255)

        self.screen = screen
        font_path = self._find_font()
        if font_path:
            self.font_large = pygame.font.Font(font_path, 32)
            self.font_medium = pygame.font.Font(font_path, 22)
            self.font_small = pygame.font.Font(font_path, 16)
        else:
            self.font_large = pygame.font.SysFont("microsoftyahei", 32)
            self.font_medium = pygame.font.SysFont("microsoftyahei", 22)
            self.font_small = pygame.font.SysFont("microsoftyahei", 16)

    def _find_font(self):
        candidates = [
            "assets/fonts/msyh.ttc",
            "assets/fonts/simhei.ttf",
            "C:/Windows/Fonts/msyh.ttc",
            "C:/Windows/Fonts/simhei.ttf",

        for path in candidates:
            if os.path.exists(path):
                return path

    def _grid_to_px(self, row, col):
        x = BOARD_X + col * CELL_SIZE
        y = BOARD_Y + row * CELL_SIZE
        return x, y

    def _px_to_grid(self, mx, my):
        col = round((mx - BOARD_X) / CELL_SIZE)
        row = round((my - BOARD_Y) / CELL_SIZE)
        if 0 <= row < BOARD_SIZE and 0 <= col < BOARD_SIZE:
            # Check if click is close enough to intersection
            gx, gy = self._grid_to_px(row, col)
            if abs(mx - gx) <= CELL_SIZE // 2 and abs(my - gy) <= CELL_SIZE // 2:
                return row, col
        return None

        for i in range(BOARD_SIZE):

            x2, y2 = self._grid_to_px(i, BOARD_SIZE - 1)
            pygame.draw.line(self.screen, COLOR_LINE, (x1, y1), (x2, y2), 1)
            # Vertical
            x1, y1 = self._grid_to_px(0, i)
            x2, y2 = self._grid_to_px(BOARD_SIZE - 1, i)
            pygame.draw.line(self.screen, COLOR_LINE, (x1, y1), (x2, y2), 1)

            x, y = self._grid_to_px(r, c)
            pygame.draw.circle(self.screen, COLOR_STAR, (x, y), 4)

    def draw_pieces(self, board, last_move=None):
```

```python
"""AI opponent using Minimax with Alpha-Beta pruning."""

import random
from board import BOARD_SIZE, EMPTY, BLACK, WHITE

# Pattern scores
SCORES = {
    'five': 1000000,
    'live_four': 100000,
    'rush_four': 10000,
    'live_three': 5000,
    'sleep_three': 500,
    'live_two': 200,
    'sleep_two': 50,
}

DIFFICULTY = {
    'easy': {'depth': 2, 'candidates': 5, 'randomize': True},
    'medium': {'depth': 4, 'candidates': 8, 'randomize': False},
    'hard': {'depth': 6, 'candidates': 15, 'randomize': False},
}

DIRECTIONS = [(0, 1), (1, 0), (1, 1), (1, -1)]


class AI:
    def __init__(self, color, difficulty='medium'):
        self.color = color
        self.opponent = WHITE if color == BLACK else BLACK
        self.config = DIFFICULTY.get(difficulty, DIFFICULTY['medium'])

    def get_move(self, board):
        candidates = self._get_candidates(board)
        if not candidates:
            return (7, 7)  # center

        if board.move_count() == 0:
            return (7, 7)

        best_score = float('-inf')
        best_moves = []

        for row, col in candidates:
            board.place(row, col, self.color)
            score = self._minimax(board, self.config['depth'] - 1, float('-inf'), float('inf'), False)
            board.undo()

            if score > best_score:
                best_score = score
                best_moves = [(row, col)]
            elif score == best_score:
                best_moves.append((row, col))

        if self.config['randomize'] and len(best_moves) > 1:
            # For easy mode, sometimes pick a suboptimal move
            pool = best_moves[:min(len(best_moves), self.config['candidates'])]
            return random.choice(pool)

        return best_moves[0]

    def _minimax(self, board, depth, alpha, beta, is_maximizing):
        # Check terminal states
        last = board.last_move()
        if last:
            winner = self._check_win_fast(board.grid, last[0], last[1])
            if winner == self.color:
                return SCORES['five'] + depth
            elif winner == self.opponent:
                return -SCORES['five'] - depth

        if depth == 0 or board.is_full():
            return self._evaluate(board)

        candidates = self._get_candidates(board)

        if is_maximizing:
            max_eval = float('-inf')
            for row, col in candidates:
                board.place(row, col, self.color)
                eval_score = self._minimax(board, depth - 1, alpha, beta, False)
                board.undo()
                max_eval = max(max_eval, eval_score)
                alpha = max(alpha, eval_score)
                if beta <= alpha:
                    break
            return max_eval
        else:
            min_eval = float('inf')
            for row, col in candidates:
                board.place(row, col, self.opponent)
                eval_score = self._minimax(board, depth - 1, alpha, beta, True)
                board.undo()
                min_eval = min(min_eval, eval_score)
                beta = min(beta, eval_score)
                if beta <= alpha:
                    break
            return min_eval

    def _get_candidates(self, board):
        """Get candidate positions near existing pieces."""
        candidates = set()
        range_limit = 2

        has_pieces = False
        for row in range(BOARD_SIZE):
            for col in range(BOARD_SIZE):
                if board.grid[row][col] != EMPTY:
                    has_pieces = True
                    for dr in range(-range_limit, range_limit + 1):
                        for dc in range(-range_limit, range_limit + 1):
                            nr, nc = row + dr, col + dc
                            if 0 <= nr < BOARD_SIZE and 0 <= nc < BOARD_SIZE:
                                if board.grid[nr][nc] == EMPTY:
                                    candidates.add((nr, nc))

        if not has_pieces:
            return [(7, 7)]

        # Score and sort candidates
        scored = []
        for row, col in candidates:
            score = self._evaluate_position(board.grid, row, col)
            scored.append((score, row, col))

        scored.sort(reverse=True)
        limit = self.config['candidates']
        return [(r, c) for _, r, c in scored[:limit]]

    def _evaluate_position(self, grid, row, col):
        """Quick evaluation of a position for candidate sorting."""
        score = 0
        for color in [self.color, self.opponent]:
            for dr, dc in DIRECTIONS:
                count = 1
                # Forward
                r, c = row + dr, col + dc
                while 0 <= r < BOARD_SIZE and 0 <= c < BOARD_SIZE and grid[r][c] == color:
                    count += 1
                    r += dr
                    c += dc
                # Backward
                r, c = row - dr, col - dc
                while 0 <= r < BOARD_SIZE and 0 <= c < BOARD_SIZE and grid[r][c] == color:
                    count += 1
                    r -= dr
                    c -= dc
                if count >= 5:
                    score += SCORES['five']
                elif count == 4:
                    score += SCORES['live_three']
                elif count == 3:
                    score += SCORES['live_two']
        return score

    def _evaluate(self, board):
        """Evaluate the entire board position."""
        ai_score = 0
        opp_score = 0

        for row in range(BOARD_SIZE):
            for col in range(BOARD_SIZE):
                color = board.grid[row][col]
                if color == EMPTY:
                    continue
                for dr, dc in DIRECTIONS:
                    # Only count in one direction to avoid double counting
                    r, c = row - dr, col - dc
                    if 0 <= r < BOARD_SIZE and 0 <= c < BOARD_SIZE and board.grid[r][c] == color:
                        continue  # skip, this is a continuation

                    count = 0
                    open_ends = 0
                    r, c = row, col
                    while 0 <= r < BOARD_SIZE and 0 <= c < BOARD_SIZE and board.grid[r][c] == color:
                        count += 1
                        r += dr
                        c += dc

                    # Check open ends
                    if 0 <= r < BOARD_SIZE and 0 <= c < BOARD_SIZE and board.grid[r][c] == EMPTY:
                        open_ends += 1
                    r2, c2 = row - dr, col - dc
                    if 0 <= r2 < BOARD_SIZE and 0 <= c2 < BOARD_SIZE and board.grid[r2][c2] == EMPTY:
                        open_ends += 1

                    pattern_score = self._pattern_score(count, open_ends)
                    if color == self.color:
                        ai_score += pattern_score
                    else:
                        opp_score += pattern_score

        return ai_score - opp_score

    def _pattern_score(self, count, open_ends):
        if count >= 5:
            return SCORES['five']
        if open_ends == 0:
            return 0
        if count == 4:
            return SCORES['live_four'] if open_ends == 2 else SCORES['rush_four']
        if count == 3:
            return SCORES['live_three'] if open_ends == 2 else SCORES['sleep_three']
        if count == 2:
            return SCORES['live_two'] if open_ends == 2 else SCORES['sleep_two']
        return 10 if open_ends == 2 else 1

    def _check_win_fast(self, grid, row, col):
        color = grid[row][col]
        if color == EMPTY:
            return None
        for dr, dc in DIRECTIONS:
            count = 1
            r, c = row + dr, col + dc
            while 0 <= r < BOARD_SIZE and 0 <= c < BOARD_SIZE and grid[r][c] == color:
                count += 1
                r += dr
                c += dc
            r, c = row - dr, col - dc
            while 0 <= r < BOARD_SIZE and 0 <= c < BOARD_SIZE and grid[r][c] == color:
                count += 1
                r -= dr
                c -= dc
            if count >= 5:
                return color
        return None
```

```python
"""Game record system: save, load, replay."""

import json
import os
from datetime import datetime

RECORDS_DIR = "records"
CLASSIC_DIR = os.path.join(RECORDS_DIR, "classic")
SAVED_DIR = os.path.join(RECORDS_DIR, "saved")


class GameRecord:
    def __init__(self):
        self.meta = {
            "date": "",
            "mode": "",
            "difficulty": "",
            "result": "",
            "total_moves": 0,
        }
        self.moves = []  # list of {"step": N, "color": "black"/"white", "pos": [r, c]}

    def add_move(self, step, color, row, col):
        color_str = "black" if color == 1 else "white"  # BLACK=1, WHITE=2
        self.moves.append({"step": step, "color": color_str, "pos": [row, col]})

    def undo_moves(self, count):
        for _ in range(count):
            if self.moves:
                self.moves.pop()

    def set_meta(self, mode, difficulty="", result=""):
        self.meta["date"] = datetime.now().strftime("%Y-%m-%d %H:%M")
        self.meta["mode"] = mode
        self.meta["difficulty"] = difficulty
        self.meta["result"] = result
        self.meta["total_moves"] = len(self.moves)

    def to_dict(self):
        self.meta["total_moves"] = len(self.moves)
        return {"meta": self.meta, "moves": self.moves}

    @classmethod
    def from_dict(cls, data):
        record = cls()
        record.meta = data.get("meta", {})
        record.moves = data.get("moves", [])
        return record

    def save(self, filename=None):
        os.makedirs(SAVED_DIR, exist_ok=True)
        if not filename:
            ts = datetime.now().strftime("%Y-%m-%d_%H-%M")
            mode = self.meta.get("mode", "game")
            filename = f"{ts}_{mode}.json"
        filepath = os.path.join(SAVED_DIR, filename)
        with open(filepath, "w", encoding="utf-8") as f:
            json.dump(self.to_dict(), f, ensure_ascii=False, indent=2)
        return filepath

    @classmethod
    def load(cls, filepath):
        with open(filepath, "r", encoding="utf-8") as f:
            data = json.load(f)
        return cls.from_dict(data)

    @classmethod
    def list_saved(cls):
        os.makedirs(SAVED_DIR, exist_ok=True)
        files = []
        for f in os.listdir(SAVED_DIR):
            if f.endswith(".json"):
                files.append(os.path.join(SAVED_DIR, f))
        files.sort(key=os.path.getmtime, reverse=True)
        return files

    @classmethod
    def list_classic(cls):
        os.makedirs(CLASSIC_DIR, exist_ok=True)
        files = []
        for f in os.listdir(CLASSIC_DIR):
            if f.endswith(".json"):
                files.append(os.path.join(CLASSIC_DIR, f))
        return files


class ReplayController:
    def __init__(self, record):
        self.record = record
        self.current_step = 0
        self.total_steps = len(record.moves)
        self.auto_play = False
        self.auto_timer = 0

    def next_step(self):
        if self.current_step < self.total_steps:
            self.current_step += 1
            return self.record.moves[self.current_step - 1]
        return None

    def prev_step(self):
        if self.current_step > 0:
            self.current_step -= 1
            return True
        return False

    def get_moves_up_to_current(self):
        return self.record.moves[:self.current_step]

    def toggle_auto_play(self):
        self.auto_play = not self.auto_play
        self.auto_timer = 0
```

```python
"""Game state machine and orchestration."""

import pygame
from board import Board, BLACK, WHITE, EMPTY
from rules import check_win, check_draw
from ai import AI
from history import GameRecord, ReplayController
from renderer import Renderer

# Game states
STATE_MENU = "menu"
STATE_PLAYING = "playing"
STATE_GAME_OVER = "game_over"
STATE_DIFFICULTY = "difficulty"
STATE_REPLAY_SELECT = "replay_select"
STATE_REPLAY = "replay"
STATE_LOAD_SELECT = "load_select"


class Game:
    def __init__(self, screen):
        self.screen = screen
        self.renderer = Renderer(screen)
        self.board = Board()
        self.state = STATE_MENU
        self.current_player = BLACK
        self.mode = None  # 'pvp' or 'pve'
        self.difficulty = None
        self.ai = None
        self.winner = None
        self.status_text = ""
        self.record = GameRecord()
        self.replay_ctrl = None
        self.replay_boards = []  # cached board states for replay

    def run(self):
        clock = pygame.time.Clock()
        running = True
        auto_play_delay = 0

        while running:
            dt = clock.tick(60)
            mouse_pos = pygame.mouse.get_pos()

            for event in pygame.event.get():
                if event.type == pygame.QUIT:
                    running = False
                    break

                if event.type == pygame.KEYDOWN:
                    if self.state == STATE_GAME_OVER:
                        if event.key == pygame.K_r:
                            self._restart()
                        elif event.key == pygame.K_m:
                            self._back_to_menu()

                if event.type == pygame.MOUSEBUTTONDOWN and event.button == 1:
                    self._handle_click(event.pos)

            # Auto-play in replay mode
            if self.state == STATE_REPLAY and self.replay_ctrl and self.replay_ctrl.auto_play:
                auto_play_delay += dt
                if auto_play_delay >= 800:  # 800ms per move
                    auto_play_delay = 0
                    move = self.replay_ctrl.next_step()
                    if move is None:
                        self.replay_ctrl.auto_play = False

            # AI move in PvE
            if self.state == STATE_PLAYING and self.mode == "pve" and self.current_player == self.ai.color:
                self._make_ai_move()

            # Draw
            self._draw(mouse_pos)
            pygame.display.flip()

        pygame.quit()

    def _handle_click(self, pos):
        if self.state == STATE_MENU:
            self._handle_menu_click(pos)
        elif self.state == STATE_DIFFICULTY:
            self._handle_difficulty_click(pos)
        elif self.state == STATE_PLAYING:
            self._handle_play_click(pos)
        elif self.state == STATE_REPLAY_SELECT:
            self._handle_replay_select_click(pos)
        elif self.state == STATE_REPLAY:
            self._handle_replay_click(pos)
        elif self.state == STATE_LOAD_SELECT:
            self._handle_load_select_click(pos)

    def _handle_menu_click(self, pos):
        buttons = self.renderer.draw_menu("五子棋", [
            ("双人对弈", "pvp"),
            ("人机对战", "pve"),
            ("棋谱回放", "replay"),
        ], pos)
        for rect, value in buttons:
            if rect.collidepoint(pos):
                if value == "pvp":
                    self._start_pvp()
                elif value == "pve":
                    self.state = STATE_DIFFICULTY
                elif value == "replay":
                    self._show_replay_list()

    def _handle_difficulty_click(self, pos):
        buttons = self.renderer.draw_menu("选择难度", [
            ("简 单", "easy"),
            ("中 等", "medium"),
            ("困 难", "hard"),
        ], pos)
        for rect, value in buttons:
            if rect.collidepoint(pos):
                self._start_pve(value)

    def _handle_play_click(self, pos):
        # Check side panel buttons first
        buttons = self._get_play_buttons()
        action = self.renderer.get_button_at(buttons, pos)
        if action == "undo":
            self._undo()
            return
        elif action == "restart":
            self._restart()
            return
        elif action == "save":
            self._save_record()
            return
        elif action == "menu":
            self._back_to_menu()
            return

        # Check board click
        grid_pos = self.renderer._px_to_grid(pos[0], pos[1])
        if grid_pos:
            row, col = grid_pos
            self._place_piece(row, col)

    def _handle_replay_select_click(self, pos):
        # Back button
        back_rect = pygame.Rect(20, 20, 80, 35)
        if back_rect.collidepoint(pos):
            self.state = STATE_MENU
            return

        records = GameRecord.list_classic() + GameRecord.list_saved()
        for i, filepath in enumerate(records):
            btn_rect = pygame.Rect(100, 100 + i * 50, 600, 40)
            if btn_rect.collidepoint(pos):
                self._start_replay(filepath)
                return

    def _handle_replay_click(self, pos):
        buttons = self.renderer.draw_replay_bar(
            self.replay_ctrl.current_step if self.replay_ctrl else 0,
            self.replay_ctrl.total_steps if self.replay_ctrl else 0
        )
        for rect, action in buttons:
            if rect.collidepoint(pos):
                if action == "prev":
                    self.replay_ctrl.prev_step()
                elif action == "next":
                    self.replay_ctrl.next_step()
                elif action == "auto":
                    self.replay_ctrl.toggle_auto_play()
                elif action == "takeover":
                    self._takeover_replay()
                return

    def _handle_load_select_click(self, pos):
        back_rect = pygame.Rect(20, 20, 80, 35)
        if back_rect.collidepoint(pos):
            self.state = STATE_MENU
            return

        records = GameRecord.list_saved()
        for i, filepath in enumerate(records):
            btn_rect = pygame.Rect(100, 100 + i * 50, 600, 40)
            if btn_rect.collidepoint(pos):
                self._load_and_resume(filepath)
                return

    def _start_pvp(self):
        self.mode = "pvp"
        self.difficulty = None
        self.board.reset()
        self.current_player = BLACK
        self.winner = None
        self.record = GameRecord()
        self.state = STATE_PLAYING
        self.status_text = "黑棋落子..."

    def _start_pve(self, difficulty):
        self.mode = "pve"
        self.difficulty = difficulty
        self.board.reset()
        self.current_player = BLACK
        self.winner = None
        self.ai = AI(WHITE, difficulty)
        self.record = GameRecord()
        self.state = STATE_PLAYING
        self.status_text = "黑棋落子..."

    def _place_piece(self, row, col):
        if not self.board.place(row, col, self.current_player):
            return

        step = self.board.move_count()
        self.record.add_move(step, self.current_player, row, col)

        # Check win
        winner = check_win(self.board.grid, row, col)
        if winner:
            self.winner = winner
            self._end_game()
            return

        # Check draw
        if check_draw(self.board.grid):
            self.winner = "draw"
            self._end_game()
            return

        # Switch player
        self.current_player = WHITE if self.current_player == BLACK else BLACK
        if self.current_player == BLACK:
            self.status_text = "黑棋落子..."
        else:
            self.status_text = "白棋落子..."

    def _make_ai_move(self):
        move = self.ai.get_move(self.board)
        if move:
            self._place_piece(move[0], move[1])

    def _undo(self):
        if self.board.move_count() == 0:
            return

        if self.mode == "pve":
            # Undo 2 moves (AI + player)
            if self.board.move_count() >= 2:
                self.board.undo()
                self.board.undo()
                self.record.undo_moves(2)
                self.current_player = BLACK
                self.status_text = "黑棋落子..."
        else:
            self.board.undo()
            self.record.undo_moves(1)
            self.current_player = WHITE if self.current_player == BLACK else BLACK
            self.status_text = "黑棋落子..." if self.current_player == BLACK else "白棋落子..."

    def _restart(self):
        if self.mode == "pvp":
            self._start_pvp()
        elif self.mode == "pve":
            self._start_pve(self.difficulty)

    def _back_to_menu(self):
        self.state = STATE_MENU
        self.board.reset()
        self.winner = None

    def _end_game(self):
        self.state = STATE_GAME_OVER
        if self.winner == "draw":
            self.status_text = "平局！"
            self.record.set_meta(self.mode, self.difficulty or "", "平局")
        elif self.winner == BLACK:
            self.status_text = "黑棋获胜！"
            self.record.set_meta(self.mode, self.difficulty or "", "黑胜")
        else:
            self.status_text = "白棋获胜！"
            self.record.set_meta(self.mode, self.difficulty or "", "白胜")

    def _save_record(self):
        filepath = self.record.save()
        self.status_text = f"已保存: {filepath}"

    def _show_replay_list(self):
        self.state = STATE_REPLAY_SELECT

    def _start_replay(self, filepath):
        record = GameRecord.load(filepath)
        self.replay_ctrl = ReplayController(record)
        self.board.reset()
        self.state = STATE_REPLAY

    def _takeover_replay(self):
        if not self.replay_ctrl:
            return
        # Rebuild board from replay moves
        self.board.reset()
        moves = self.replay_ctrl.get_moves_up_to_current()
        for move in moves:
            color = BLACK if move["color"] == "black" else WHITE
            r, c = move["pos"]
            self.board.place(r, c, color)

        self.mode = "pvp"
        self.current_player = BLACK
        if len(moves) % 2 == 0:
            self.current_player = BLACK
        else:
            self.current_player = WHITE
        self.winner = None
        self.record = GameRecord()
        for move in moves:
            color = BLACK if move["color"] == "black" else WHITE
            self.record.add_move(move["step"], color, move["pos"][0], move["pos"][1])
        self.state = STATE_PLAYING
        self.status_text = "黑棋落子..." if self.current_player == BLACK else "白棋落子..."

    def _load_and_resume(self, filepath):
        record = GameRecord.load(filepath)
        self.board.reset()
        for move in record.moves:
            color = BLACK if move["color"] == "black" else WHITE
            r, c = move["pos"]
            self.board.place(r, c, color)
        self.record = record
        self.mode = "pvp"
        self.current_player = BLACK if len(record.moves) % 2 == 0 else WHITE
        self.winner = None
        self.state = STATE_PLAYING
        self.status_text = "黑棋落子..." if self.current_player == BLACK else "白棋落子..."

    def _get_play_buttons(self):
        return [
            ("悔 棋", "undo"),
            ("重新开始", "restart"),
            ("保存棋谱", "save"),
            ("返回菜单", "menu"),
        ]

    def _draw(self, mouse_pos):

            self.renderer.draw_menu("五子棋", [
                ("双人对弈", "pvp"),
                ("人机对战", "pve"),

            ], mouse_pos)

        elif self.state == STATE_DIFFICULTY:

                ("简 单", "easy"),
                ("中 等", "medium"),
                ("困 难", "hard"),
            ], mouse_pos)

        elif self.state == STATE_PLAYING:
            self.renderer.draw_board()
            self.renderer.draw_pieces(self.board, self.board.last_move())
            mode_text = "双人对弈" if self.mode == "pvp" else f"人机-{self.difficulty}"
            self.renderer.draw_top_bar(self.current_player, mode_text)
            self.renderer.draw_bottom_bar(self.status_text, self.board.move_count(), mode_text)

        elif self.state == STATE_GAME_OVER:
            self.renderer.draw_board()
            self.renderer.draw_pieces(self.board, self.board.last_move())
            mode_text = "双人对弈" if self.mode == "pvp" else f"人机-{self.difficulty}"
            self.renderer.draw_top_bar(self.current_player, mode_text)
            self.renderer.draw_bottom_bar(self.status_text, self.board.move_count(), mode_text)
            self.renderer.draw_side_panel(self._get_play_buttons(), self.board.move_count(), mode_text)
```
